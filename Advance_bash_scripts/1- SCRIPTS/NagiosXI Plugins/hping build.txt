#CENTOS


yum -y install libpcap-devel tcl tcl-devel gcc
git clone https://github.com/antirez/hping.git

mkdir -p /usr/local/include/net
ln -sf /usr/include/pcap-bpf.h /usr/local/include/net/bpf.h

cd hping3
./configure
make
make install
